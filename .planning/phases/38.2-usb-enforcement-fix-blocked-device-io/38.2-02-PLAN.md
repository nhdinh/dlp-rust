```rust
    let before: HashSet<char> = detector.blocked_drives.read().iter().copied().collect();
    let mut now_present: HashSet<char> = HashSet::new();

            }

        }
    }
}
```

### From RESEARCH.md — WR-01 fix approach

When USB_DEVICE arrives, scan `blocked_drives` for a drive letter that is:
- In `blocked_drives` (known USB)
- NOT in `device_identities` (no identity yet)
- Is a removable drive (just to be safe)

If exactly one such drive letter exists, assign the identity to it and call `apply_tier_enforcement`. If zero or multiple, park in `pending_identity` as before.

### From RESEARCH.md — startup gap fix approach

Add a `scan_existing_usb_identities` method to `UsbDetector` that:
1. Enumerates already-present USB mass storage devices via SetupDi (`GUID_DEVINTERFACE_USB_DEVICE` with `DIGCF_PRESENT`)
2. For each USB device found, parses VID/PID/serial from the device path
3. Finds the matching drive letter by checking which drive letter in `blocked_drives` has no identity yet
4. Populates `device_identities` and calls `apply_tier_enforcement`

Alternatively (simpler): use the existing `dlp_common::usb::enumerate_connected_usb_devices` to get identities, then match against `blocked_drives`.

    ```rust
    fn on_usb_device_arrival(detector: &UsbDetector, device_path: &str) {
        let mut identity = parse_usb_device_path(device_path);
        identity.description = setupdi_description_for_device(device_path);

        // WR-01 fix: If VOLUME arrived before USB_DEVICE, the drive letter is already
        // in blocked_drives but device_identities has no entry. Check for exactly one
        // un-identity'd USB drive and reconcile immediately.
        let unmapped_drives: Vec<char> = {
            let blocked = detector.blocked_drives.read();
            let identities = detector.device_identities.read();
            blocked
                .iter()
                .filter(|letter| !identities.contains_key(letter))
                .copied()
                .collect()
        };

        if unmapped_drives.len() == 1 {
            let letter = unmapped_drives[0];
            info!(
                vid = %identity.vid,
                pid = %identity.pid,
                serial = %identity.serial,
                drive = %letter,
                "USB identity reconciled with existing volume (VOLUME-arrived-first race)"
            );
            detector.device_identities.write().insert(letter, identity.clone());
            apply_tier_enforcement(letter, &identity);
            // Clear any stale pending_identity — this device is now fully reconciled.
            *detector.pending_identity.lock() = None;
            return;
        }

        // Normal path: park identity for reconciliation when VOLUME arrives later.
        info!(
            vid = %identity.vid,
            pid = %identity.pid,
            serial = %identity.serial,
            description = %identity.description,
            unmapped_count = unmapped_drives.len(),
            "USB device arrived — identity parked, awaiting VOLUME notification"
        );
        *detector.pending_identity.lock() = Some(identity);
    }
    ```

    Key design decisions:
    1. The heuristic only fires when EXACTLY ONE unmapped drive exists. If zero, the VOLUME hasn't arrived yet — park as before. If multiple, we cannot disambiguate — park as before (safer than guessing wrong).
    2. The `blocked_drives` set is the source of truth for "this drive letter is a USB mass storage device" — it was populated either by `scan_existing_drives` at startup or by `handle_volume_event` when the VOLUME arrived.
    3. After reconciliation, `pending_identity` is explicitly cleared to prevent double-processing when a subsequent VOLUME event arrives.
    4. The `return` after reconciliation prevents falling through to the pending_identity assignment.

    Implementation approach (simpler, no new Win32 calls):
    ```rust
    /// Attempts to capture USB device identities for drives already in
    /// `blocked_drives` and applies tier enforcement.
    ///
    /// Called once at startup after `scan_existing_drives`. Uses SetupDi to
    /// enumerate present USB devices, then matches them against unmapped drive
    /// letters using the same heuristic as the WR-01 fix.
    #[cfg(windows)]
    pub fn scan_existing_usb_identities(&self) {
        use dlp_common::usb::enumerate_connected_usb_devices;

        let identities = match enumerate_connected_usb_devices() {
            Ok(ids) => ids,
            Err(e) => {
                warn!(error = %e, "failed to enumerate connected USB devices at startup");
                return;
            }
        };

        for identity in identities {
            // Skip devices that already have an identity mapped (e.g., from a
            // prior scan or a race with an arrival notification).
            let already_mapped = {
                let ids = self.device_identities.read();
                ids.values().any(|id| {
                    id.vid == identity.vid && id.pid == identity.pid && id.serial == identity.serial
                })
            };
            if already_mapped {
                continue;
            }

            // Find unmapped drives using the same heuristic as WR-01.
            let unmapped: Vec<char> = {
                let blocked = self.blocked_drives.read();
                let ids = self.device_identities.read();
                blocked
                    .iter()
                    .filter(|letter| !ids.contains_key(letter))
                    .copied()
                    .collect()
            };

            if unmapped.len() == 1 {
                let letter = unmapped[0];
                info!(
                    vid = %identity.vid,
                    pid = %identity.pid,
                    serial = %identity.serial,
                    drive = %letter,
                    "startup scan: USB identity reconciled with existing drive"
                );
                self.device_identities.write().insert(letter, identity.clone());
                apply_tier_enforcement(letter, &identity);
            } else if unmapped.len() > 1 {
                // Ambiguous: multiple unmapped USB drives. We cannot determine which
                // identity belongs to which drive letter. Park the identity in
                // pending_identity — if a VOLUME notification arrives later, it will
                // be reconciled. But pending_identity is a single slot, so we can only
                // park one. Pick the first identity and log a warning.
                warn!(
                    vid = %identity.vid,
                    pid = %identity.pid,
                    serial = %identity.serial,
                    unmapped_count = unmapped.len(),
                    "startup scan: multiple unmapped USB drives — cannot disambiguate identity"
                );
                // Only park if slot is empty.
                let mut pending = self.pending_identity.lock();
                if pending.is_none() {
                    *pending = Some(identity.clone());
                }
            } else {
                // No unmapped drives — this USB device may not have a volume yet
                // (e.g., a device with no mounted partition). Log and skip.
                debug!(
                    vid = %identity.vid,
                    pid = %identity.pid,
                    serial = %identity.serial,
                    "startup scan: USB device has no mapped drive letter — skipping"
                );
            }
        }
    }
    ```

// Payload for POST/PUT /policies:
pub struct PolicyPayload {
    pub id: String,
    pub name: String,
    pub description: Option<String>,
    pub priority: u32,
    pub conditions: serde_json::Value,
    pub action: String,
    pub enabled: bool,
}

// Response shape for GET/POST/PUT /policies:
pub struct PolicyResponse {
    pub id: String,
    pub name: String,
    pub description: Option<String>,
    pub priority: u32,
    pub conditions: serde_json::Value,
    pub action: String,
    pub enabled: bool,
    pub version: i64,
    pub updated_at: String,
}

// create_policy returns 201 Created. update_policy returns 200 OK.
// delete_policy returns 204 No Content. Both update and delete return
// 404 NotFound if the policy id is unknown.

```rust
//! Repository for the `managed_origins` table.
//!
//! Encapsulates SQL for managed web origin CRUD. Managed origins are URL-pattern
//! strings (e.g. `"https://company.sharepoint.com/*"`) used by the Chrome
//! Enterprise Connector (Phase 29) and surfaced in the admin TUI (Phase 28).

use rusqlite::params;

use crate::db::{Pool, UnitOfWork};

/// Plain data row returned by managed origins reads.
#[derive(Debug, Clone)]
pub struct ManagedOriginRow {
    /// Server-generated UUID string (primary key).
    pub id: String,
    /// URL pattern string, e.g. `"https://company.sharepoint.com/*"`.
    pub origin: String,
}

/// Stateless repository for the `managed_origins` table.
pub struct ManagedOriginsRepository;

impl ManagedOriginsRepository {
    /// Returns all managed origin entries ordered by `rowid` ascending.
    pub fn list_all(pool: &Pool) -> rusqlite::Result<Vec<ManagedOriginRow>> {
        let conn = pool
            .get()
            .map_err(|e| rusqlite::Error::ToSqlConversionFailure(Box::new(e)))?;
        let mut stmt = conn.prepare(
            "SELECT id, origin FROM managed_origins ORDER BY rowid ASC",
        )?;
        let rows = stmt.query_map([], |row| {
            Ok(ManagedOriginRow {
                id: row.get(0)?,
                origin: row.get(1)?,
            })
        })?;
        rows.collect()
    }

    /// Inserts a new managed origin row. Returns `rusqlite::Error` on UNIQUE
    /// constraint violation (duplicate origin string).
    pub fn insert(uow: &UnitOfWork<'_>, row: &ManagedOriginRow) -> rusqlite::Result<()> {
        uow.tx.execute(
            "INSERT INTO managed_origins (id, origin) VALUES (?1, ?2)",
            params![row.id, row.origin],
        )?;
        Ok(())
    }

    /// Deletes the managed origin entry with the given `id`.
    /// Returns the number of rows deleted (0 if `id` did not exist).
    pub fn delete_by_id(uow: &UnitOfWork<'_>, id: &str) -> rusqlite::Result<usize> {
        uow.tx
            .execute("DELETE FROM managed_origins WHERE id = ?1", params![id])
    }
}
```

```rust
/// `GET /admin/managed-origins` — unauthenticated; returns all managed origins.
///
/// Used by the Phase 29 Chrome Enterprise Connector agent to poll the list,
/// and by the admin TUI to populate the managed origins list screen.
async fn list_managed_origins_handler(
    State(state): State<Arc<AppState>>,
) -> Result<impl IntoResponse, AppError> {
    let pool = Arc::clone(&state.pool);
    let rows = tokio::task::spawn_blocking(move || ManagedOriginsRepository::list_all(&pool))
        .await
        .map_err(|e| AppError::Internal(format!("spawn_blocking join: {e}")))?
        .map_err(|e| AppError::Internal(format!("db list_all: {e}")))?;
    let resp: Vec<ManagedOriginResponse> = rows
        .into_iter()
        .map(|r| ManagedOriginResponse { id: r.id, origin: r.origin })
        .collect();
    Ok(Json(resp))
}

/// `POST /admin/managed-origins` — JWT-protected; inserts a new origin.
///
/// Returns 409 Conflict if the origin string already exists (UNIQUE constraint).
async fn create_managed_origin_handler(
    State(state): State<Arc<AppState>>,
    Json(req): Json<ManagedOriginRequest>,
) -> Result<impl IntoResponse, AppError> {
    let id = uuid::Uuid::new_v4().to_string();
    let row = ManagedOriginRow { id: id.clone(), origin: req.origin.clone() };
    let pool = Arc::clone(&state.pool);
    let result = tokio::task::spawn_blocking(move || {
        let mut conn = pool
            .get()
            .map_err(|e| AppError::Internal(format!("pool get: {e}")))?;
        let uow = crate::db::UnitOfWork::new(&mut *conn)
            .map_err(|e| AppError::Internal(format!("begin tx: {e}")))?;
        ManagedOriginsRepository::insert(&uow, &row).map_err(|e| {
            // SQLite error code 2067 = SQLITE_CONSTRAINT_UNIQUE
            if let rusqlite::Error::SqliteFailure(ref fe, _) = e {
                if fe.extended_code == 2067 {
                    return AppError::Conflict("origin already exists".to_string());
                }
            }
            AppError::Internal(format!("db insert: {e}"))
        })?;
        uow.commit()
            .map_err(|e| AppError::Internal(format!("commit: {e}")))?;
        Ok::<(), AppError>(())
    })
    .await
    .map_err(|e| AppError::Internal(format!("spawn_blocking join: {e}")))?;
    result?;
    Ok((StatusCode::OK, Json(ManagedOriginResponse { id, origin: req.origin })))
}

/// `DELETE /admin/managed-origins/{id}` — JWT-protected; removes by UUID.
///
/// Returns 204 No Content on success, 404 Not Found if id does not exist.
async fn delete_managed_origin_handler(
    State(state): State<Arc<AppState>>,
    Path(id): Path<String>,
) -> Result<impl IntoResponse, AppError> {
    let pool = Arc::clone(&state.pool);
    let id_clone = id.clone();
    let affected = tokio::task::spawn_blocking(move || {
        let mut conn = pool
            .get()
            .map_err(|e| AppError::Internal(format!("pool get: {e}")))?;
        let uow = crate::db::UnitOfWork::new(&mut *conn)
            .map_err(|e| AppError::Internal(format!("begin tx: {e}")))?;
        let n = ManagedOriginsRepository::delete_by_id(&uow, &id_clone)
            .map_err(|e| AppError::Internal(format!("db delete: {e}")))?;
        uow.commit()
            .map_err(|e| AppError::Internal(format!("commit: {e}")))?;
        Ok::<usize, AppError>(n)
    })
    .await
    .map_err(|e| AppError::Internal(format!("spawn_blocking join: {e}")))?;
    match affected? {
        0 => Err(AppError::NotFound(format!("managed origin {id} not found"))),
        _ => Ok(StatusCode::NO_CONTENT),
    }
}
```

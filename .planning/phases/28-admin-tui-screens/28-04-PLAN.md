```rust
fn action_load_managed_origin_list(app: &mut App) {
    match app.client.get::<Vec<serde_json::Value>>("/admin/managed-origins") {
        Ok(origins) => app.screen = Screen::ManagedOriginList { origins, selected: 0 },
        Err(e) => {
            app.status_message = Some((
                format!("Error loading origins: {e}"),
                StatusKind::Error,
            ));
        }
    }
}
```

3. Add `handle_managed_origin_list` function:
```rust
fn handle_managed_origin_list(app: &mut App, key: KeyEvent) {
    let (origins_len, selected_idx) = match &app.screen {
        Screen::ManagedOriginList { origins, selected } => (origins.len(), *selected),
        _ => return,
    };
    match key.code {
        KeyCode::Up | KeyCode::Down => {
            if origins_len == 0 { return; }
            if let Screen::ManagedOriginList { selected, .. } = &mut app.screen {
                nav(selected, origins_len, key.code);
            }
        }
        KeyCode::Char('a') => {
            app.screen = Screen::TextInput {
                prompt: "Origin URL pattern (e.g. https://company.sharepoint.com/*):".to_string(),
                input: String::new(),
                purpose: InputPurpose::AddManagedOrigin,
            };
        }
        KeyCode::Char('d') => {
            if origins_len == 0 { return; }
            let (id, origin_str) = match &app.screen {
                Screen::ManagedOriginList { origins, selected } => {
                    let id = origins[*selected]["id"].as_str().unwrap_or("").to_string();
                    let o = origins[*selected]["origin"].as_str().unwrap_or("").to_string();
                    (id, o)
                }
                _ => return,
            };
            if id.is_empty() { return; }
            app.screen = Screen::Confirm {
                message: format!("Remove origin '{origin_str}'?"),
                yes_selected: true,
                purpose: ConfirmPurpose::DeleteManagedOrigin { id },
            };
        }
        // Return to DevicesMenu with "Managed Origins" item (index 1) highlighted.
        KeyCode::Esc => app.screen = Screen::DevicesMenu { selected: 1 },
        _ => {}
    }
}
```

4. In `handle_text_input` InputPurpose match, add:
```rust
InputPurpose::AddManagedOrigin => {
    let origin = input.trim().to_string();
    if origin.is_empty() {
        app.status_message = Some((
            "Origin cannot be empty.".to_string(),
            StatusKind::Error,
        ));
        action_load_managed_origin_list(app);
        return;
    }
    let body = serde_json::json!({ "origin": origin });
    match app.client.post::<serde_json::Value>("/admin/managed-origins", &body) {
        Ok(_) => {
            app.status_message = Some((
                "Origin added.".to_string(),
                StatusKind::Success,
            ));
            action_load_managed_origin_list(app);
        }
        Err(e) => {
            app.status_message = Some((
                format!("Error adding origin: {e}"),
                StatusKind::Error,
            ));
            action_load_managed_origin_list(app);
        }
    }
}
```

5. In `handle_confirm` ConfirmPurpose match, add:

```rust
ConfirmPurpose::DeleteManagedOrigin { id } => {

            ));
            app.screen = Screen::DevicesMenu { selected: 1 };
        }
    }
}
```

mod detective_tc {
    use dlp_agent::clipboard::ContentClassifier;
    use dlp_agent::interception::PolicyMapper;
    use dlp_common::audit::{AuditEvent, EventType};
    use dlp_common::{Action, Classification, Decision};

    /// TC-80: Confidential file access → logged (detective)
    #[test]
    fn test_tc_80_confidential_access_logged() {
        // T3 access is allowed but must emit EventType::Access (not Block).
        let cls = PolicyMapper::provisional_classification(r"C:\Confidential\report.xlsx");
        assert_eq!(cls, Classification::T3);
        assert!(cls.is_sensitive());

        // Verify the expected audit event shape for detective logging.
        let event = AuditEvent::new(
            EventType::Access,
            "S-1-5-21-TEST".into(),
            "testuser".into(),
            r"C:\Confidential\report.xlsx".into(),
            Classification::T3,
            Action::READ,
            Decision::ALLOW, // detective allows but logs
            "AGENT-TC80".into(),
            1,
        );
        assert_eq!(event.event_type, EventType::Access);
        assert_eq!(event.decision, Decision::ALLOW);
        assert_eq!(event.classification, Classification::T3);
    }

    /// TC-81: Bulk download Confidential → alert (detective)
    #[test]
    fn test_tc_81_bulk_download_confidential_alert() {
        // Validates the threshold concept: classify_text results must all be T3+.
        let sensitive_texts = vec![
            "CONFIDENTIAL: Q1 financials",
            "Card: 4111111111111111",
            "SSN: 123-45-6789",
        ];
        let classified: Vec<_> = sensitive_texts
            .iter()
            .map(|t| ContentClassifier::classify(t))
            .collect();
        assert!(
            classified.iter().all(|c| *c >= Classification::T3),
            "all texts must be T3+ for bulk detection"
        );
        // Acceptance: 10 unique T3+ file accesses in 60s → EventType::Alert.
        todo!("TC-81: bulk download (10 files/60s) — threshold detection not implemented")
    }

    /// TC-82: Restricted access outside working hours → alert (detective)

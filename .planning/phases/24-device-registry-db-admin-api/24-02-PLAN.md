**list_device_registry_handler** (unauthenticated GET):
```rust
/// Returns all registered USB device trust-tier entries.
///
/// `GET /admin/device-registry` — intentionally unauthenticated so agents can
/// poll without stored credentials (D-01, per 24-CONTEXT.md).
async fn list_device_registry_handler(
    State(state): State<Arc<AppState>>,
) -> Result<Json<Vec<DeviceRegistryResponse>>, AppError> {
    let pool = Arc::clone(&state.pool);
    let rows = tokio::task::spawn_blocking(move || {
        repositories::DeviceRegistryRepository::list_all(&pool)
    })
    .await
    .map_err(|e| AppError::Internal(e.to_string()))?
    .map_err(|e| AppError::Internal(e.to_string()))?;
    let response: Vec<DeviceRegistryResponse> = rows.into_iter().map(Into::into).collect();
    Ok(Json(response))
}
```

**upsert_device_registry_handler** (JWT-protected POST):
```rust
/// Registers a new device or updates trust_tier/description for an existing one.
///
/// `POST /admin/device-registry` — requires JWT Bearer auth (D-02).
/// Upserts on (vid, pid, serial) conflict: the UUID is preserved, only
/// trust_tier and description are updated (D-04).
///
/// # Errors
///
/// Returns 422 if `trust_tier` is not one of "blocked", "read_only", "full_access".
async fn upsert_device_registry_handler(
    State(state): State<Arc<AppState>>,
    Json(body): Json<DeviceRegistryRequest>,
) -> Result<Json<DeviceRegistryResponse>, AppError> {
    const VALID_TIERS: &[&str] = &["blocked", "read_only", "full_access"];
    if !VALID_TIERS.contains(&body.trust_tier.as_str()) {
        return Err(AppError::UnprocessableEntity(format!(
            "invalid trust_tier '{}'; must be one of: blocked, read_only, full_access",
            body.trust_tier
        )));
        // NOTE: If AppError::UnprocessableEntity does not exist, use whichever
        // AppError variant maps to StatusCode::UNPROCESSABLE_ENTITY (422).
        // Check lib.rs or the AppError definition before writing.
    }
    let id = uuid::Uuid::new_v4().to_string();
    let created_at = chrono::Utc::now().to_rfc3339();
    let row = repositories::DeviceRegistryRow {
        id: id.clone(),
        vid: body.vid,
        pid: body.pid,
        serial: body.serial,
        description: body.description,
        trust_tier: body.trust_tier,
        created_at,
    };
    let pool = Arc::clone(&state.pool);
    let returned_row = tokio::task::spawn_blocking(move || {
        let conn = pool.get().map_err(|e| rusqlite::Error::ToSqlConversionFailure(Box::new(e)))?;
        let uow = crate::db::UnitOfWork::new(&conn)?;
        repositories::DeviceRegistryRepository::upsert(&uow, &row)?;
        uow.commit()?;
        // Re-read the row to return the persisted state (id may differ on conflict update).
        repositories::DeviceRegistryRepository::list_all(&pool)
    })
    .await
    .map_err(|e| AppError::Internal(e.to_string()))?
    .map_err(|e| AppError::Internal(e.to_string()))?;
    // Find the upserted row by (vid, pid, serial) — the id in DB may be the original on conflict.
    // Simpler: return the row with matching id from the full list, or re-query by vid/pid/serial.
    // Practical approach: after upsert, list_all and find by (vid, pid, serial). The upsert
    // returns the EXISTING row's id on conflict. The simplest correct approach is:
    // after upsert succeeds, query SELECT ... WHERE vid=? AND pid=? AND serial=?.
    // Implement this as a DeviceRegistryRepository::get_by_device_key helper if needed,
    // or inline the query.
    let _ = returned_row; // Remove if using list approach above.
    // IMPLEMENTATION NOTE: The cleaner approach is to add a
    // DeviceRegistryRepository::get_by_device_key(pool, vid, pid, serial) method
    // that does SELECT ... WHERE vid=?1 AND pid=?2 AND serial=?3 — add this method
    // to device_registry.rs and use it here instead of post-upsert list scan.
    todo!() // Replace with actual implementation once get_by_device_key is added.
}
```

**delete_device_registry_handler** (JWT-protected DELETE):
```rust
/// Removes a registered device entry by its server-generated UUID.
///
/// `DELETE /admin/device-registry/{id}` — requires JWT Bearer auth (D-02, D-06).
/// Returns 204 on success, 404 if the UUID does not exist.
async fn delete_device_registry_handler(
    State(state): State<Arc<AppState>>,
    Path(id): Path<String>,
) -> Result<StatusCode, AppError> {
    let pool = Arc::clone(&state.pool);
    let rows_deleted = tokio::task::spawn_blocking(move || {
        let conn = pool.get().map_err(|e| rusqlite::Error::ToSqlConversionFailure(Box::new(e)))?;
        let uow = crate::db::UnitOfWork::new(&conn)?;
        let n = repositories::DeviceRegistryRepository::delete_by_id(&uow, &id)?;
        uow.commit()?;
        rusqlite::Result::Ok(n)
    })
    .await
    .map_err(|e| AppError::Internal(e.to_string()))?
    .map_err(|e| AppError::Internal(e.to_string()))?;

    if rows_deleted == 0 {
        return Err(AppError::NotFound("device not found".to_string()));
    }
    Ok(StatusCode::NO_CONTENT)
}
```

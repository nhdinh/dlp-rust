pub struct PolicyRow { pub id: String, pub name: String, ... }

impl PolicyRepository {
    pub fn list(pool: &Pool) -> rusqlite::Result<Vec<PolicyRow>> {
        let conn = pool.get().map_err(|e| rusqlite::Error::ToSqlConversionFailure(Box::new(e)))?;
        let mut stmt = conn.prepare("SELECT ... FROM policies ORDER BY priority ASC")?;
        let rows = stmt.query_map([], |row| Ok(PolicyRow { ... }))?;
        rows.collect()
    }
    pub fn insert(uow: &UnitOfWork<'_>, record: &PolicyRow) -> rusqlite::Result<()> {
        uow.tx.execute("INSERT INTO policies ...", params![...])?;
        Ok(())
    }
    pub fn delete(uow: &UnitOfWork<'_>, id: &str) -> rusqlite::Result<usize> {
        uow.tx.execute("DELETE FROM policies WHERE id = ?1", params![id])
    }
}
```

**Row types:**

use crate::db::{Pool, UnitOfWork};
use rusqlite::params;

/// Plain data row returned by device registry reads.
#[derive(Debug, Clone)]
pub struct DeviceRegistryRow {
    /// Server-generated UUID string (primary key).
    pub id: String,
    /// USB Vendor ID hex string, e.g. "0951".
    pub vid: String,
    /// USB Product ID hex string, e.g. "1666".
    pub pid: String,
    /// Device serial number, or "(none)" for devices without one.
    pub serial: String,
    /// Human-readable device description from USB descriptor. Empty string if unknown.
    pub description: String,
    /// Trust tier: "blocked", "read_only", or "full_access".
    pub trust_tier: String,
    /// ISO-8601 timestamp of when this entry was created.
    pub created_at: String,
}
```

**Repository struct** (stateless, no fields):

pub struct DeviceRegistryRepository;

**Methods:**

`list_all(pool: &Pool) -> rusqlite::Result<Vec<DeviceRegistryRow>>`
- SELECT id, vid, pid, serial, description, trust_tier, created_at FROM device_registry ORDER BY created_at ASC
- Map rows to DeviceRegistryRow, collect

`upsert(uow: &UnitOfWork<'_>, row: &DeviceRegistryRow) -> rusqlite::Result<()>`
- Use INSERT INTO device_registry (id, vid, pid, serial, description, trust_tier, created_at) VALUES (?1,?2,?3,?4,?5,?6,?7) ON CONFLICT(vid, pid, serial) DO UPDATE SET trust_tier = excluded.trust_tier, description = excluded.description
- (Note: ON CONFLICT DO UPDATE is supported by rusqlite/SQLite 3.24+; this is the idiomatic upsert for named conflicts. Do NOT use INSERT OR REPLACE as it changes the primary key UUID on conflict.)

`delete_by_id(uow: &UnitOfWork<'_>, id: &str) -> rusqlite::Result<usize>`
- DELETE FROM device_registry WHERE id = ?1
- Return rows_affected (0 if not found)

```rust
/// Policy record returned by `GET /admin/policies` (server's PolicyResponse shape).
///
/// This is the authoritative schema for import/export JSON. The JSON object stored in
/// `Screen::ImportConfirm.policies` is deserialized as `serde_json::Value` at parse time
/// for flexibility, then converted to this struct before POST/PUT.
///
/// Fields map directly to `PolicyResponse` from `dlp-server/src/admin_api.rs` §116–§135.
#[derive(Debug, Clone, serde::Deserialize)]
pub struct PolicyResponse {
    pub id: String,
    pub name: String,
    pub description: Option<String>,
    pub priority: u32,
    /// JSON-encoded conditions array (`Vec<PolicyCondition>` on the wire).
    pub conditions: serde_json::Value,
    pub action: String,
    pub enabled: bool,
    #[serde(default)]
    pub version: i64,
    #[serde(default)]
    pub updated_at: String,
}

/// Payload for `POST /admin/policies` and `PUT /admin/policies/{id}`.
///
/// Matches `PolicyPayload` from `dlp-server/src/admin_api.rs` §97–§112.
/// Dropped from `PolicyResponse`: `version`, `updated_at`.
#[derive(Debug, Clone, serde::Serialize)]
pub struct PolicyPayload {
    pub id: String,
    pub name: String,
    pub description: Option<String>,
    pub priority: u32,
    pub conditions: serde_json::Value,
    pub action: String,
    pub enabled: bool,
}

impl From<PolicyResponse> for PolicyPayload {
    fn from(r: PolicyResponse) -> Self {
        Self {
            id: r.id,
            name: r.name,
            description: r.description,
            priority: r.priority,
            conditions: r.conditions,
            action: r.action,
            enabled: r.enabled,
        }
    }
}
```

**Note:** `serde_json::Value` is already in scope via the existing `serde_json` import in
the crate (used throughout `app.rs` for `serde_json::Value` types in `Screen` variants).
</action>

```rust
/// Handles key events for the `Screen::ImportConfirm` variant.
///
/// Navigation: Up/Down cycles only between rows 3 ([Confirm]) and 4 ([Cancel]).
/// Enter on row 3 → execute import (POST new policies, PUT conflicting policies).
/// Enter on row 4 / Esc → return to PolicyMenu.
///
/// Import execution (D-09 / D-11 / D-17 / D-18 / D-19 from Phase 17 context):
/// - POST non-conflicting policies (IDs not on server).
/// - PUT conflicting policies (IDs already on server).
/// - Abort on first failure with per-policy error message.
/// - Transitions to ImportState::Success or ImportState::Error.
fn handle_import_confirm(app: &mut App, key: KeyEvent) {
    use crate::app::{ImportCaller, ImportState};

    let caller = match &app.screen {
        Screen::ImportConfirm { caller, .. } => *caller,
        _ => return,
    };

    // If not Pending, only Esc/Enter to dismiss is allowed.
    if !matches!(
        app.screen,
        Screen::ImportConfirm {
            state: ImportState::Pending,
            ..
        }
    ) {
        match key.code {
            KeyCode::Enter | KeyCode::Esc => {
                let return_to = match caller {
                    ImportCaller::PolicyMenu => Screen::PolicyMenu { selected: 0 },
                };
                app.screen = return_to;
            }
            _ => {}
        }
        return;
    }

    match key.code {
        KeyCode::Up | KeyCode::Down => {
            if let Screen::ImportConfirm { selected, .. } = &mut app.screen {
                *selected = if *selected == 3 { 4 } else { 3 };
            }
        }
        KeyCode::Enter => {
            let return_to = match caller {
                ImportCaller::PolicyMenu => Screen::PolicyMenu { selected: 0 },
            };

            // Extract state needed for import execution.
            let (policies, existing_ids) = match &app.screen {
                Screen::ImportConfirm {
                    policies,
                    existing_ids,
                    ..
                } => (policies.clone(), existing_ids.clone()),
                _ => return,
            };

            // Transition to InProgress immediately so the UI reflects working state.
            if let Screen::ImportConfirm { state, .. } = &mut app.screen {
                *state = ImportState::InProgress;
            }
            drop(return_to); // avoid borrow conflict below

            // Build existing ID set for conflict diff.
            let existing_set: std::collections::HashSet<String> =
                existing_ids.into_iter().collect();

            // Partition: non-conflicting → POST, conflicting → PUT.
            let (to_create, to_update): (Vec<_>, Vec<_>) = policies
                .into_iter()
                .filter_map(|v| serde_json::from_value::<crate::app::PolicyResponse>(v).ok())
                .partition(|p| !existing_set.contains(&p.id));

            let mut created = 0usize;
            let mut updated = 0usize;

            // POST non-conflicting policies (D-09).
            for policy in to_create {
                let payload: crate::app::PolicyPayload = policy.clone().into();
                let result = app.rt.block_on(
                    app.client.post::<serde_json::Value, _>("policies", &payload),
                );
                match result {
                    Ok(_) => {
                        created += 1;
                    }
                    Err(e) => {
                        // Abort on first failure (D-11 / D-19).
                        if let Screen::ImportConfirm { state, .. } = &mut app.screen {
                            *state = ImportState::Error(format!(
                                "Failed on policy '{}': {e}",
                                policy.name
                            ));
                        }
                        return;
                    }
                }
            }

            // PUT conflicting policies (D-09).
            for policy in to_update {
                let payload: crate::app::PolicyPayload = policy.clone().into();
                let path = format!("policies/{}", policy.id);
                let result =
                    app.rt.block_on(app.client.put::<serde_json::Value, _>(&path, &payload));
                match result {
                    Ok(_) => {
                        updated += 1;
                    }
                    Err(e) => {
                        // Abort on first failure (D-11 / D-19).
                        if let Screen::ImportConfirm { state, .. } = &mut app.screen {
                            *state = ImportState::Error(format!(
                                "Failed on policy '{}': {e}",
                                policy.name
                            ));
                        }
                        return;
                    }
                }
            }

            // All succeeded — show summary (D-20).
            if let Screen::ImportConfirm { state, .. } = &mut app.screen {
                *state = ImportState::Success { created, updated };
            }
        }
        KeyCode::Esc => {
            let return_to = match caller {
                ImportCaller::PolicyMenu => Screen::PolicyMenu { selected: 0 },
            };
            app.screen = return_to;
        }
        _ => {}
    }
}
```

**Key design decisions in this implementation:**
- `serde_json::from_value::<PolicyResponse>(v)` converts the raw JSON values
  stored in `Screen::ImportConfirm.policies` to typed structs before wire conversion.
- The `existing_ids` stored in `ImportConfirm` is already the server-ID list fetched
  in `action_import_policies` (Wave 1), so no second GET is needed.
- `IntoIterator` is used on `Vec<serde_json::Value>` via `.into_iter()` to feed
  `.filter_map()` — no intermediate allocation.
- `HashSet` lookup is O(1) per policy.
- Import aborts on the **first** POST or PUT failure (D-11 / D-19).
</action>

```rust
#[cfg(test)]
mod import_export_tests {
    use super::*;

    #[test]
    fn policy_response_to_payload_drops_version_and_updated_at() {
        let response = PolicyResponse {
            id: "test-id".to_string(),
            name: "Test Policy".to_string(),
            description: Some("A test".to_string()),
            priority: 10,
            conditions: serde_json::json!([]),
            action: "DENY".to_string(),
            enabled: true,
            version: 42,
            updated_at: "2026-01-01T00:00:00Z".to_string(),
        };

        let payload: PolicyPayload = response.into();

        assert_eq!(payload.id, "test-id");
        assert_eq!(payload.name, "Test Policy");
        assert_eq!(payload.description, Some("A test".to_string()));
        assert_eq!(payload.priority, 10);
        assert_eq!(payload.conditions, serde_json::json!([]));
        assert_eq!(payload.action, "DENY");
        assert_eq!(payload.enabled, true);
        // version and updated_at are not fields on PolicyPayload — verified by the
        // fact that the struct compiles without those fields.
    }

    #[test]
    fn policy_response_deserializes_from_server_json() {
        let json = serde_json::json!({
            "id": "policy-abc",
            "name": "Block T4 Exports",
            "description": "Prevent T4 data leaving the network",
            "priority": 1,
            "conditions": [
                { "attribute": "Classification", "operator": "eq", "value": "T4" }
            ],
            "action": "DENY",
            "enabled": true,
            "version": 3,
            "updated_at": "2026-04-01T12:00:00Z"
        });

        let response: PolicyResponse = serde_json::from_value(json).expect("must parse");
        assert_eq!(response.id, "policy-abc");
        assert_eq!(response.name, "Block T4 Exports");
        assert_eq!(response.priority, 1);
        assert_eq!(response.action, "DENY");
        assert!(response.enabled);
        assert!(response.description.is_some());
        assert_eq!(response.conditions.as_array().unwrap().len(), 1);
    }

    #[test]
    fn policy_response_missing_optional_fields_default() {
        // version and updated_at are optional with #[serde(default)].
        let json = serde_json::json!({
            "id": "minimal",
            "name": "Minimal",
            "priority": 5,
            "conditions": [],
            "action": "ALLOW",
            "enabled": false
        });

        let response: PolicyResponse = serde_json::from_value(json).expect("must parse");
        assert_eq!(response.version, 0);
```

    **Step B -- Mock backend.**
    ```rust
    /// Deterministic in-memory backend. Each call to `query_volume` consumes the
    /// next value from a per-drive-letter script. `read_boot_status_registry`
    /// returns a configured fixed value or error.
    struct MockBackend {
        scripts: Mutex<std::collections::HashMap<char, Vec<Result<(EncryptionStatus, Option<EncryptionMethod>), EncryptionError>>>>,
        boot_registry: Mutex<Result<u32, EncryptionError>>,
        call_count: AtomicUsize,
    }

    impl MockBackend {
        fn new() -> Arc<Self> {
            Arc::new(Self {
                scripts: Mutex::new(std::collections::HashMap::new()),
                boot_registry: Mutex::new(Err(EncryptionError::RegistryReadFailed("unset".into()))),
                call_count: AtomicUsize::new(0),
            })
        }
        fn script(self: &Arc<Self>, letter: char, results: Vec<Result<(EncryptionStatus, Option<EncryptionMethod>), EncryptionError>>) {
            self.scripts.lock().insert(letter, results);
        }
        fn calls(&self) -> usize { self.call_count.load(Ordering::SeqCst) }
    }

    impl EncryptionBackend for MockBackend {
        fn query_volume(&self, drive_letter: char) -> Result<(EncryptionStatus, Option<EncryptionMethod>), EncryptionError> {
            self.call_count.fetch_add(1, Ordering::SeqCst);
            let mut scripts = self.scripts.lock();
            let queue = scripts.entry(drive_letter).or_default();
            if queue.is_empty() {
                return Err(EncryptionError::VolumeNotFound);
            }
            queue.remove(0)
        }
        fn read_boot_status_registry(&self) -> Result<u32, EncryptionError> {
            match &*self.boot_registry.lock() {
                Ok(v) => Ok(*v),
                Err(e) => Err(EncryptionError::RegistryReadFailed(e.to_string())),
            }
        }
    }
    ```

    /// First-set-wins on `set_disk_enumerator` is fine because subsequent tests
    /// reuse the SAME enumerator and overwrite its RwLock contents through this fn.
    fn seed_enumerator(disks: Vec<DiskIdentity>, mark_complete: bool) -> Arc<DiskEnumerator> {
        let enumerator = match dlp_agent::detection::disk::get_disk_enumerator() {
            Some(e) => e,
            None => {
                let fresh = Arc::new(DiskEnumerator::new());
                set_disk_enumerator(Arc::clone(&fresh));
                fresh
            }
        };
        {
            let mut w = enumerator.discovered_disks.write();
            *w = disks.clone();
        }
        {
            let mut id_map = enumerator.instance_id_map.write();
            id_map.clear();
            for d in &disks {
                id_map.insert(d.instance_id.clone(), d.clone());
            }
        }
        {
            let mut letter_map = enumerator.drive_letter_map.write();
            letter_map.clear();
            for d in &disks {

                    letter_map.insert(l, d.clone());
                }
            }
        }
        *enumerator.enumeration_complete.write() = mark_complete;
        enumerator

    /// Reset the EncryptionChecker singleton's interior state so each test starts
    /// from a fresh checker without contending with `OnceLock::set` (which is one-shot).
